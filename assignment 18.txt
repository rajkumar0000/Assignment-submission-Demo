//  #7. Write a function to check whether a given string is palindrome or not.

#include<stdio.h>
#include<string.h>
int main()
{
    int i,j,lnth,count=0;
    char  a[100],b[100];
    printf("Enter string : ");
    fgets(a,100,stdin);
    lnth=strlen(a);
    j=lnth-2;
    for(i=0;i<lnth-1;i++)
    {
        b[j]=a[i];
        j--;
    }
    for(i=0;i<lnth-1;i++)
    {
        if(a[i]!=b[i])
            count++;
    }

    if(count==0)
        printf("Yes , This is palintrome string ");
    else
        printf("No , This is not palintrome string ");

    return 0;
}

//  #9. Write a function to reverse a string word wise. (For example if the given string is
//     “Mysirg Education Services” then the resulting string should be “Services Education Mysirg” )


#include<stdio.h>
#include<string.h>
void swep(char a[],int s,int e)
{
    for(int i=s;i<e;i++)
    {
        char temp=a[i];
        a[i]=a[e];
        a[e]=temp;
        e--;
    }
}
int main()
{
    int i=0,lnth,end=0,start=-1;
    char  a[30]={" Mysirg Education Services "};
    lnth=strlen(a);
    for(i=0;i<lnth;i++)
    {
       if(a[i]==' ')
       {
           swep(a,start,end);
           start=end;
       }
        end++;
    }
    strrev(a);
    printf("%s",a);

    return 0;
}
