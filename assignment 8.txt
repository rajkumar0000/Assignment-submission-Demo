* * * * * *

*/

#include<stdio.h>

{

//  #6. Write a program to draw the following patterns
/*

* * * * * * * * *

  * * * * * * *

    * * * * *

      * * *

        *

*/

#include<stdio.h>
main()
{
    int i,j,k,l,n=5;
    for(i=0;i<n;i++)
    {
        for(j=0;j<i;j++)
            printf("  ");
        for(k=n;k>j;k--)
            printf("* ");
        for(l=n;l>i+1;l--)
            printf("* ");


        printf("\n\n");
    }
}

//  Write a program to draw the following patterns:
/*  #18.
     *
    ***
   *****
  *******
 *********
  *******
   *****
    ***
     *
*/

#include<stdio.h>
main()
{
    int i,j,n=5;
    for(i=0;i<n;i++)
    {
        for(j=n;j>i;j--)
            printf(" ");
        for(j=0;j<=i;j++)
            printf("*");
        for(j=0;j<i;j++)
            printf("*");

        printf("\n");
    }

    for(i=0;i<n;i++)
    {
        for(j=0;j<=i+1;j++)
            printf(" ");
        for(j=n-1;j>i;j--)
            printf("*");
        for(j=n-2;j>i;j--)
            printf("*");

    printf("\n");
    }
}
