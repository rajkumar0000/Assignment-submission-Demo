//  #1. Write a program to calculate sum of first N natural numbers

#include<stdio.h>
main()
{
    int i,n,num=0;
    printf("Enter a number ");
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    {
        num=num+i;
    }
    printf("%d ",num);
}
